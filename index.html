Swift Code Example

import Foundation

// Sample Swift function
func calculateSum(_ numbers: [Int]) -> Int {
    return numbers.reduce(0, +)
}

let numbers = [1, 2, 3, 4, 5]
let sum = calculateSum(numbers)
print("The sum is: \(sum)")
